\begin{lstlisting}
SELECT 
	TeileNr, Farbe, MIN(Preis) AS niedrigster_preis, 
	MAX(Preis) AS hoechster_preis, 
	MAX(Preis) - MIN(Preis) AS Differenz 
FROM 
	Liefert 
WHERE 
	LiefNr <> 0
GROUP BY 
	TeileNr, Farbe
